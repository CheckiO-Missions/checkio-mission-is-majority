Da Python True als 1 und False als 0 interpretiert, können Sie einfach die Summe der Elemente von items finden und diese sollte größer als die Hälfte der Länge sein. 

def is_majority(items: list[bool]) -> bool:
    
    return sum(items)> len(items) / 2
